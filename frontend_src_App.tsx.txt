import React, { useState } from 'react'
import SearchBar from './components/SearchBar'
import ProductList, { Product } from './components/ProductList'
import { fetchProducts } from './api/api'

const App: React.FC = () => {
  const [products, setProducts] = useState<Product[]>([])
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const onSearch = async (query: string) => {
    setLoading(true)
    setError(null)
    try {
      const res = await fetchProducts(query)
      setProducts(res)
    } catch (e: any) {
      setError(e.message || 'An error occurred')
    } finally {
      setLoading(false)
    }
  }

  return (
    <div className="min-h-screen flex flex-col items-center p-6">
      <header className="w-full max-w-4xl">
        <h1 className="text-3xl font-semibold mb-4">The AI's Mercado Libre â€” Frontend</h1>
        <p className="text-sm text-gray-600 mb-6">
          Simple demo frontend to search products and show predictions from the backend.
        </p>
      </header>

      <main className="w-full max-w-4xl bg-white rounded-lg shadow p-6">
        <SearchBar onSearch={onSearch} />
        {loading && <p className="mt-4 text-gray-500">Loading...</p>}
        {error && <p className="mt-4 text-red-500">Error: {error}</p>}
        <ProductList products={products} />
      </main>

      <footer className="w-full max-w-4xl mt-6 text-xs text-gray-500">
        <p>Set API base url with <code>VITE_API_URL</code> environment variable.</p>
      </footer>
    </div>
  )
}

export default App
